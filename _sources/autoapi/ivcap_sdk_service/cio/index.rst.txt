      :param name: Optional name. Defaults to None.

      :type metadata: Optional[MetaDict | List[MetaDict]], optional
      :param seekable: If true, writable should be seekable (needed for NetCDF). Defaults to False.
      :type seekable: bool, optional
      :param on_close: Called with assigned artifact ID. Defaults to None.
      :type on_close: Optional[OnCloseF], optional

      :returns: A file-like object to write deliver artifact content - needs to be closed
      :rtype: IOWritable
